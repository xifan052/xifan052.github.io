实现一个promise、promise.all()


class MyPromise {
  constructor(executor) {
    this.status = 'pending';
    this.value = undefined;
    this.callbacks = [];

    const resolve = (value) => {
      if (this.status === 'pending') {
        this.status = 'fulfilled';
        this.value = value;
        this.callbacks.forEach((callback) => {
          callback.onFulfilled(value);
        });
      }
    };

    const reject = (reason) => {
      if (this.status === 'pending') {
        this.status = 'rejected';
        this.value = reason;
        this.callbacks.forEach((callback) => {
          callback.onRejected(reason);
        });
      }
    };

    try {
      executor(resolve, reject);
    } catch (error) {
      reject(error);
    }
  }

  then(onFulfilled, onRejected) {
    return new MyPromise((resolve, reject) => {
      if (this.status === 'pending') {
        this.callbacks.push({
          onFulfilled: (value) => {
            try {
              if (typeof onFulfilled === 'function') {
                const result = onFulfilled(value);
                resolve(result);
              } else {
                resolve(value);
              }
            } catch (error) {
              reject(error);
            }
          },
          onRejected: (reason) => {
            if (typeof onRejected === 'function') {
              try {
                const result = onRejected(reason);
                resolve(result);
              } catch (error) {
                reject(error);
              }
            } else {
              reject(reason);
            }
          },
        });
      } else if (this.status === 'fulfilled') {
        try {
          if (typeof onFulfilled === 'function') {
            const result = onFulfilled(this.value);
            resolve(result);
          } else {
            resolve(this.value);
          }
        } catch (error) {
          reject(error);
        }
      } else if (this.status === 'rejected') {
        if (typeof onRejected === 'function') {
          try {
            const result = onRejected(this.value);
            resolve(result);
          } catch (error) {
            reject(error);
          }
        } else {
          reject(this.value);
        }
      }
    });
  }
}


MyPromise.all = function (promises) {
  return new MyPromise((resolve, reject) => {
    let results = [];
    let completedPromises = 0;

    promises.forEach((promise, index) => {
      promise.then(
        (value) => {
          results[index] = value;
          completedPromises++;
          if (completedPromises === promises.length) {
            resolve(results);
          }
        },
        (reason) => {
          reject(reason);
        }
      );
    });
  });
};

实现一个 instanceOf 方法

function myInstanceOf(obj, constructor) {
    // 判断参数是否合法
    if (obj === null || typeof obj !== 'object' || typeof constructor !== 'function') {
        return false;
    }
    
    let proto = Object.getPrototypeOf(obj); // 获取对象的原型

    while (proto) {
        // 判断当前原型是否等于指定构造函数的原型
        if (proto === constructor.prototype) {
            return true;
        }
        // 继续沿着原型链向上查找
        proto = Object.getPrototypeOf(proto);
    }

    return false; // 遍历完整条原型链仍未找到，返回 false
}

// 测试示例
function Person(name) {
    this.name = name;
}

const person = new Person('Alice');
console.log(myInstanceOf(person, Person)); // true
console.log(myInstanceOf(person, Object)); // true，因为所有对象都是 Object 的实例
console.log(myInstanceOf(person, Array)); // false

手写一个发布订阅

实现一个 Emitter 类，具备下面的使用方式 // const emitter = new Emitter(); // const sub1 = emitter.on('click', callback); // const sub2 = emitter.on('click', callback); // emitter.emit('click', '1', '2'); // sub1.off('click')

class EventBus {
  constructor() {
    this.events = {}; // 存储事件及其回调函数
  }

  // 订阅事件
  on(event, callback) {
    if (!this.events[event]) {
      this.events[event] = []; // 如果事件还没有订阅过，初始化一个空数组
    }
    this.events[event].push(callback); // 添加回调函数到事件数组中
  }

  // 取消订阅事件
  off(event, callback) {
    if (!this.events[event]) return; // 如果事件没有订阅过，直接返回

    // 过滤出不需要移除的回调函数
    this.events[event] = this.events[event].filter(cb => cb !== callback);
  }

  // 发布事件
  emit(event, ...args) {
    if (!this.events[event]) return; // 如果没有订阅该事件，直接返回

    // 调用事件的所有回调函数
    this.events[event].forEach(callback => callback(...args));
  }

  // 只触发一次的订阅
  once(event, callback) {
    const onceCallback = (...args) => {
      callback(...args); // 调用原始回调
      this.off(event, onceCallback); // 触发一次后取消订阅
    };
    this.on(event, onceCallback); // 将一次性回调添加到事件订阅中
  }
}

const bus = new EventBus();

// 订阅事件
bus.on('event1', (data) => {
  console.log('Event 1 received:', data);
});

// 触发事件
bus.emit('event1', { message: 'Hello, EventBus!' }); // 输出: Event 1 received: { message: 'Hello, EventBus!' }

// 订阅一次性事件
bus.once('event2', (data) => {
  console.log('Event 2 received only once:', data);
});

// 触发一次性事件
bus.emit('event2', { message: 'This will be logged once.' }); // 输出: Event 2 received only once: { message: 'This will be logged once.' }

// 再次触发一次性事件，不会再触发
bus.emit('event2', { message: 'This will NOT be logged.' });

// 取消订阅
const callback = (data) => console.log('Event 3 received:', data);
bus.on('event3', callback);
bus.emit('event3', { message: 'This will be logged.' }); // 输出: Event 3 received: { message: 'This will be logged.' }
bus.off('event3', callback);
bus.emit('event3', { message: 'This will NOT be logged.' }); // 不会输出任何内容

    function flat(arr) {
        let result = [];
        arr.forEach(item => {
            if (Array.isArray(item)) {
                result = result.concat(flat(item));
            } else {
                result.push(item);
            }
        });
        return result;
    }

格式化字符串 "1,222,333"

    function formatNumberWithCommas(str) {
        const reversedStr = str.split('').reverse().join(''); // 反转字符串便于处理
        let formattedStr = '';
        for (let i = 0; i < reversedStr.length; i++) {
            if (i > 0 && i % 3 === 0) {
                formattedStr += ',';  // 每隔三位插入逗号
            }
            formattedStr += reversedStr[i];
        }
        return formattedStr.split('').reverse().join('');  // 最后再次反转字符串得到最终结果
    }

    const str = '1222332313123';
    const formattedStr = formatNumberWithCommas(str);
    console.log(formattedStr); // 输出 "1,222,333"

手写防抖和节流

防抖是指在事件被触发后，等待一定时间间隔，如果在这个时间间隔内没有再次触发该事件，则执行相应操作。如果在这个时间间隔内再次触发了事件，则重新等待一定时间。

    function debounce(func, delay) {
        let timer;
        return function (...args) {
            clearTimeout(timer);
            timer = setTimeout(() => {
                func.apply(this, args);
            }, delay);
        };
    }

    // 示例用法
    function handleInput() {
        console.log('Input event debounced');
    }

    const debouncedHandleInput = debounce(function () {
        console.log('Input event debounced');
    }, 300);

    // 监听输入框的输入事件，并进行防抖处理
    document.addEventListener("mousemove", debouncedHandleInput);

节流是指在一定时间间隔内只触发一次事件，即使这段时间内触发了多次。它可以用于限制事件处理函数的执行频率。

    function throttle(func, limit) {
        let inThrottle;
        return function (...args) {
            if (!inThrottle) {
                func.apply(this, args);
                inThrottle = true;
                setTimeout(() => inThrottle = false, limit);
            }
        };
    }

    // 示例用法
    function handleScroll() {
        console.log('Scroll event throttled');
    }

    const throttledHandleScroll = throttle(handleScroll, 100);

    // 监听滚动事件，并进行节流处理
    window.addEventListener('scroll', throttledHandleScroll);

版本号的排序


// 比较两个版本号字符串的函数
function compareVersions(version1, version2) {
    const v1 = version1.split('.').map(Number); // 将版本号字符串拆分成数字数组
    const v2 = version2.split('.').map(Number);

    const maxLength = Math.max(v1.length, v2.length); // 获取数字数组的最大长度

    for (let i = 0; i < maxLength; i++) {
        const num1 = i < v1.length ? v1[i] : 0; // 获取数字数组中对应位置的数字，若超出索引则默认为0
        const num2 = i < v2.length ? v2[i] : 0;

        if (num1 < num2) { // 比较对应位置的数字大小
            return -1;
        } else if (num1 > num2) {
            return 1;
        }
    }

    return 0; // 版本号相等
}

// 版本号数组
const versions = ['1.11.2', '1.9.9', '1.12.0', '1.10.0'];

// 对版本号数组进行排序
versions.sort(compareVersions);

console.log(versions); // 输出排序后的版本号数组

字符串中出现最多次数的字符


function findMostFrequentChar(str) {
    const charCount = {};

    // 统计每个字符的出现次数
    for (let char of str) {
        if (charCount[char]) {
            charCount[char]++;
        } else {
            charCount[char] = 1;
        }
    }

    let maxCount = 0;
    let mostFrequentChar = '';

    // 寻找出现次数最多的字符
    for (let char in charCount) {
        if (charCount[char] > maxCount) {
            maxCount = charCount[char];
            mostFrequentChar = char;
        }
    }

    return mostFrequentChar;
}

// 示例用法
const str = 'abcaabbc';
const mostFrequentChar = findMostFrequentChar(str);
console.log(mostFrequentChar); // 输出 'a'，因为字符 'a' 在字符串中出现了最多次数

找连续数中缺失的数

function findMissingNumber(arr) {
  // 计算数组中所有数字的总和
  const sum = arr.reduce((accumulator, currentValue) => accumulator + currentValue, 0);

  // 计算从最小值到最大值的连续整数序列的总和
  // 使用等差数列求和公式：(n * (n + 1)) / 2
  const min = Math.min(...arr);
  const max = Math.max(...arr);
  const expectedSum = ((max - min) / 2 + 1) * min + ((max - min) / 2) * max;

  // 缺失的数字是两个总和的差
  return expectedSum - sum;
}

// 示例
const numbers = [1, 2, 4, 5]; // 假设缺失的数字是 3
console.log(findMissingNumber(numbers)); // 输出: 3

给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串 s ，判断字符串是否有效

有效字符串需满足：

1.左括号必须用相同类型的右括号闭合。 2.左括号必须以正确的顺序闭合。 3.每个右括号都有一个对应的相同类型的左括号。

示例 1：

输入：s = "()" 输出：true

示例 2：

输入：s = "{[()]}()" 输出：true

示例 3：

输入：s = "(]" 输出：false

要判断给定的字符串 s 是否是一个有效的括号字符串，我们可以使用一个栈（Stack）的数据结构来解决这个问题。栈用于跟踪最近遇到的尚未匹配的左括号。以下是使用 JavaScript 实现的示例代码：

function isValid(s) {
    let stack = [];
    let mapping = {
        ')': '(',
        '}': '{',
        ']': '['
    };

    for (let i = 0; i < s.length; i++) {
        let char = s[i];
        if (char === '(' || char === '{' || char === '[') {
            // 如果是左括号，压入栈中
            stack.push(char);
        } else {
            // 如果是右括号
            if (stack.length === 0) {
                // 栈为空，说明没有对应的左括号
                return false;
            }
            // 弹出栈顶元素
            let topElement = stack.pop();
            // 检查是否匹配
            if (mapping[char] !== topElement) {
                return false;
            }
        }
    }

    // 如果栈为空，则所有括号都正确匹配
    return stack.length === 0;
}

// 示例
console.log(isValid("()")); // true
console.log(isValid("{[()]}")); // true
console.log(isValid("(]")); // false

这段代码中，我们首先定义了一个映射对象 mapping 来存储每种右括号对应的左括号。然后遍历字符串 s，对于遇到的每个字符：

  • 如果是左括号，我们将其压入栈中。
  • 如果是右括号，我们检查栈顶是否有对应的左括号。如果有，我们将其与栈顶元素进行比较，如果匹配则弹出栈顶元素；如果不匹配或栈为空，则字符串无效。

最后，如果遍历结束后栈为空，则说明所有括号都正确匹配，字符串有效；否则，字符串无效。
